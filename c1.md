'''
#include "raylib.h"
#include <stdlib.h>

// 화면 및 게임 설정 상수
#define SCREEN_WIDTH 800
#define SCREEN_HEIGHT 600
#define BRICK_ROWS 6
#define BRICK_COLS 10
#define INITIAL_LIVES 5

// 블록 구조체
typedef struct Brick {
    Rectangle rect;
    Color color;
    bool active;
} Brick;

int main(void) {
    // 1. 창 초기화
    InitWindow(SCREEN_WIDTH, SCREEN_HEIGHT, "Rainbow Brick Breaker");
    SetTargetFPS(60);

    // 2. 게임 오브젝트 상태 초기화
    // 패들 (플레이어 바)
    Rectangle paddle = { SCREEN_WIDTH / 2 - 60, SCREEN_HEIGHT - 40, 120, 15 };
    float paddleSpeed = 8.0f;

    // 공
    Vector2 ballPosition = { SCREEN_WIDTH / 2, SCREEN_HEIGHT / 2 };
    Vector2 ballSpeed = { 5.0f, -5.0f };
    float ballRadius = 8.0f;
    bool ballActive = false; // 스페이스바를 누르면 시작

    // 무지개 색상 배열 (6줄)
    Color rainbowColors[BRICK_ROWS] = {
        RED, ORANGE, YELLOW, GREEN, BLUE, PURPLE
    };

    // 블록 배열 생성 및 초기화
    Brick bricks[BRICK_ROWS][BRICK_COLS];
    int brickWidth = 74;
    int brickHeight = 20;
    int startX = 30;
    int startY = 50;
    int padding = 4;

    for (int i = 0; i < BRICK_ROWS; i++) {
        for (int j = 0; j < BRICK_COLS; j++) {
            bricks[i][j].rect = (Rectangle){
                startX + j * (brickWidth + padding),
                startY + i * (brickHeight + padding),
                brickWidth,
                brickHeight
            };
            bricks[i][j].color = rainbowColors[i]; // 줄마다 무지개 색상 부여
            bricks[i][j].active = true;
        }
    }

    // 게임 점수 및 목숨
    int lives = INITIAL_LIVES;
    int score = 0;
    bool gameOver = false;
    bool gameWon = false;

    // 3. 메인 게임 루프
    while (!WindowShouldClose()) {
        // --- 업데이트 (Update) ---
        if (!gameOver && !gameWon) {
            // 패들 이동 제어
            if (IsKeyDown(KEY_LEFT) && paddle.x > 0) paddle.x -= paddleSpeed;
            if (IsKeyDown(KEY_RIGHT) && paddle.x < SCREEN_WIDTH - paddle.width) paddle.x += paddleSpeed;

            // 공 시작 제어 (스페이스바)
            if (!ballActive && IsKeyPressed(KEY_SPACE)) {
                ballActive = true;
            }

            // 공 움직임 처리
            if (ballActive) {
                ballPosition.x += ballSpeed.x;
                ballPosition.y += ballSpeed.y;

                // 벽 충돌 (좌/우)
                if (ballPosition.x - ballRadius <= 0 || ballPosition.x + ballRadius >= SCREEN_WIDTH) {
                    ballSpeed.x *= -1.0f;
                }
                // 벽 충돌 (상단)
                if (ballPosition.y - ballRadius <= 0) {
                    ballSpeed.y *= -1.0f;
                }
                // 바닥으로 떨어졌을 때 (실패)
                if (ballPosition.y + ballRadius >= SCREEN_HEIGHT) {
                    lives--;
                    ballActive = false;
                    // 공과 패들 위치 리셋
                    ballPosition = (Vector2){ SCREEN_WIDTH / 2, SCREEN_HEIGHT / 2 };
                    ballSpeed = (Vector2){ 5.0f, -5.0f };
                    paddle.x = SCREEN_WIDTH / 2 - paddle.width / 2;

                    if (lives <= 0) {
                        gameOver = true;
                    }
                }

                // 패들과 공 충돌 체크
                if (CheckCollisionCircleRec(ballPosition, ballRadius, paddle)) {
                    if (ballSpeed.y > 0) {
                        ballSpeed.y *= -1.0f;
                        // 충돌 위치에 따라 반사 각도 약간 조절
                        ballSpeed.x = (ballPosition.x - (paddle.x + paddle.width / 2)) / (paddle.width / 2) * 5.0f;
                    }
                }

                // 블록과 공 충돌 체크
                int activeBricks = 0;
                for (int i = 0; i < BRICK_ROWS; i++) {
                    for (int j = 0; j < BRICK_COLS; j++) {
                        if (bricks[i][j].active) {
                            activeBricks++;
                            if (CheckCollisionCircleRec(ballPosition, ballRadius, bricks[i][j].rect)) {
                                bricks[i][j].active = false;
                                ballSpeed.y *= -1.0f; // 충돌 시 튕김
                                score += 10;
                                break;
                            }
                        }
                    }
                }
                
                // 모든 블록을 다 깨면 승리
                if (activeBricks == 0) {
                    gameWon = true;
                }
            }
        } else {
            // 게임오버나 승리 상태에서 R 키를 누르면 재시작
            if (IsKeyPressed(KEY_R)) {
                lives = INITIAL_LIVES;
                score = 0;
                gameOver = false;
                gameWon = false;
                ballActive = false;
                ballPosition = (Vector2){ SCREEN_WIDTH / 2, SCREEN_HEIGHT / 2 };
                paddle.x = SCREEN_WIDTH / 2 - paddle.width / 2;
                
                for (int i = 0; i < BRICK_ROWS; i++) {
                    for (int j = 0; j < BRICK_COLS; j++) {
                        bricks[i][j].active = true;
                    }
                }
            }
        }

        // --- 화면 그리기 (Draw) ---
        BeginDrawing();
        ClearBackground(BLACK); // 배경은 검은색으로 무지개색 강조

        // 블록 그리기
        for (int i = 0; i < BRICK_ROWS; i++) {
            for (int j = 0; j < BRICK_COLS; j++) {
                if (bricks[i][j].active) {
                    DrawRectangleRec(bricks[i][j].rect, bricks[i][j].color);
                    // 블록 테두리를 살짝 그려서 구분감 주기
                    DrawRectangleLinesEx(bricks[i][j].rect, 1, RAYWHITE);
                }
            }
        }

        // 패들과 공 그리기
        DrawRectangleRec(paddle, LIGHTGRAY);
        DrawCircleV(ballPosition, ballRadius, WHITE);

        // UI 정보 표시 (점수 및 목숨)
        DrawText(TextFormat("SCORE: %04d", score), 20, 15, 20, WHITE);
        DrawText(TextFormat("LIVES: %d", lives), SCREEN_WIDTH - 120, 15, 20, RED);

        // 게임 시작 전 안내 문구
        if (!ballActive && !gameOver && !gameWon) {
            DrawText("PRESS SPACE TO START", SCREEN_WIDTH / 2 - 140, SCREEN_HEIGHT / 2 + 50, 20, MAROON);
        }

        // 게임오버 화면
        if (gameOver) {
            DrawText("GAME OVER", SCREEN_WIDTH / 2 - 140, SCREEN_HEIGHT / 2 - 20, 50, RED);
            DrawText("PRESS 'R' TO RESTART", SCREEN_WIDTH / 2 - 130, SCREEN_HEIGHT / 2 + 40, 20, LIGHTGRAY);
        }

        // 게임 승리 화면
        if (gameWon) {
            DrawText("YOU WIN!", SCREEN_WIDTH / 2 - 120, SCREEN_HEIGHT / 2 - 20, 50, GREEN);
            DrawText("PRESS 'R' TO PLAY AGAIN", SCREEN_WIDTH / 2 - 140, SCREEN_HEIGHT / 2 + 40, 20, LIGHTGRAY);
        }

        EndDrawing();
    }
'''
    // 4. 자원 해제 및 종료
    CloseWindow();

    return 0;
}
